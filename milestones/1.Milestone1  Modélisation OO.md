## Concepts POO mis en œuvre

| Concept | Où | Ce que ça fait |
|---------|-----|----------------|
| `class` + `__init__` | `Produit`, `Mouvement` | Définit la structure d'un objet |
| Encapsulation (`_`) | `StockService._produits` | Données internes non modifiables directement |
| Méthodes d'instance | `produit.est_en_alerte()` | Comportement attaché à l'objet |
| Méthodes de classe | `Produit.from_dict()` | Constructeur alternatif |
| Séparation des couches | `models/` · `services/` · `ui/` | Chaque couche a une seule responsabilité |

### Étape 1 — Les modèles (20 min)

La classe `Produit` représente un article du stock. Elle est au cœur de toute l'application.

```python
# models/produit.py
class Produit:
    """Un article commercialisé par Al Qalam."""

    def __init__(self, ref: str, nom: str, categorie: str,
                 prix_achat: float, prix_vente: float,
                 qte: int, seuil_min: int = 5):
        self.ref        = ref
        self.nom        = nom
        self.categorie  = categorie
        self.prix_achat = prix_achat
        self.prix_vente = prix_vente
        self.qte        = qte
        self.seuil_min  = seuil_min

    # ── Méthodes métier ───────────────────────────────────────────────────
    def est_en_alerte(self) -> bool:
        """True si la quantité est sous le seuil minimum."""
        return self.qte < self.seuil_min

    def valeur_stock(self) -> float:
        """Valeur d'inventaire : quantité × prix d'achat."""
        return self.qte * self.prix_achat

    def marge_unitaire(self) -> float:
        """Marge brute par unité vendue."""
        return self.prix_vente - self.prix_achat

    # ── Sérialisation JSON ────────────────────────────────────────────────
    def to_dict(self) -> dict:
        """Convertit le produit en dictionnaire pour la sauvegarde JSON."""
        return {
            "ref": self.ref, "nom": self.nom, "categorie": self.categorie,
            "prix_achat": self.prix_achat, "prix_vente": self.prix_vente,
            "qte": self.qte, "seuil_min": self.seuil_min,
        }

    @classmethod
    def from_dict(cls, d: dict) -> "Produit":
        """Crée un Produit depuis un dictionnaire (chargement JSON)."""
        return cls(
            ref=d["ref"], nom=d["nom"], categorie=d["categorie"],
            prix_achat=d["prix_achat"], prix_vente=d["prix_vente"],
            qte=d["qte"], seuil_min=d.get("seuil_min", 5),
        )
```

La classe `Mouvement` trace chaque entrée ou sortie de stock :

```python
# models/mouvement.py
from datetime import datetime

class Mouvement:
    """Enregistrement d'une opération de stock (entrée ou sortie)."""

    TYPES_VALIDES = {"entree", "sortie"}

    def __init__(self, ref_produit: str, type_mvt: str,
                 qte: int, note: str = ""):
        if type_mvt not in self.TYPES_VALIDES:
            raise ValueError(f"type_mvt doit être 'entree' ou 'sortie', reçu : {type_mvt}")
        if qte <= 0:
            raise ValueError(f"qte doit être positive, reçu : {qte}")

        self.ref_produit = ref_produit
        self.type_mvt    = type_mvt
        self.qte         = qte
        self.note        = note
        self.date        = datetime.now().isoformat()

    def est_entree(self) -> bool:
        return self.type_mvt == "entree"

    def est_sortie(self) -> bool:
        return self.type_mvt == "sortie"

    def to_dict(self) -> dict:
        return {
            "ref_produit": self.ref_produit,
            "type_mvt"   : self.type_mvt,
            "qte"        : self.qte,
            "note"       : self.note,
            "date"       : self.date,
        }
```

### Étape 2 — StockService (20 min)

`StockService` est le **cerveau** de l'application : il maintient le catalogue en mémoire et persiste les données.

```python
# services/stock_service.py
import json
from pathlib import Path
from models.produit   import Produit
from models.mouvement import Mouvement
from config           import DATA_DIR

class StockService:
    """Gère l'ensemble du stock : ajout, entrée, sortie, persistance."""

    def __init__(self):
        self._produits   = {}   # { ref: Produit } — accès O(1) par ref
        self._mouvements = []   # liste chronologique des mouvements
        self._chemin     = DATA_DIR / "stock.json"
        self.charger()          # chargement automatique au démarrage

    def ajouter_produit(self, produit: Produit) -> None:
        if produit.ref in self._produits:
            raise ValueError(f"La référence '{produit.ref}' existe déjà.")
        self._produits[produit.ref] = produit
        self.sauvegarder()

    def entree_stock(self, ref: str, qte: int, note: str = "") -> None:
        if qte <= 0:
            raise ValueError("La quantité doit être positive.")
        produit = self.get_produit(ref)
        produit.qte += qte
        self._mouvements.append(Mouvement(ref, "entree", qte, note))
        self.sauvegarder()

    def sortie_stock(self, ref: str, qte: int, note: str = "") -> None:
        if qte <= 0:
            raise ValueError("La quantité doit être positive.")
        produit = self.get_produit(ref)
        if produit.qte < qte:
            raise ValueError(
                f"Stock insuffisant : {produit.qte} disponible, {qte} demandé."
            )
        produit.qte -= qte
        self._mouvements.append(Mouvement(ref, "sortie", qte, note))
        self.sauvegarder()

    def get_produit(self, ref: str) -> Produit:
        if ref not in self._produits:
            raise KeyError(f"Produit '{ref}' introuvable.")
        return self._produits[ref]

    def lister_tous(self) -> list:
        return list(self._produits.values())

    def produits_en_alerte(self) -> list:
        alertes = []
        for p in self._produits.values():
            if p.est_en_alerte():
                alertes.append(p)
        return alertes

    def sauvegarder(self) -> None:
        DATA_DIR.mkdir(parents=True, exist_ok=True)
        data = [p.to_dict() for p in self._produits.values()]
        self._chemin.write_text(
            json.dumps(data, indent=2, ensure_ascii=False),
            encoding="utf-8"
        )

    def charger(self) -> None:
        if not self._chemin.exists():
            self._seeder()
            return
        try:
            data = json.loads(self._chemin.read_text(encoding="utf-8"))
            self._produits = {d["ref"]: Produit.from_dict(d) for d in data}
        except (json.JSONDecodeError, KeyError):
            self._seeder()

    def _seeder(self) -> None:
        """10 produits de démonstration — certains volontairement en rupture."""
        demo = [
            Produit("CRAY-001", "Crayon HB",     "Écriture", 0.15, 0.50, 150, 20),
            Produit("CRAY-002", "Crayon 2B",      "Écriture", 0.20, 0.60,   8, 20),
            Produit("STYL-001", "Stylo Bleu",     "Écriture", 0.30, 0.90, 200, 30),
            Produit("STYL-002", "Stylo Rouge",    "Écriture", 0.30, 0.90,   4, 30),
            Produit("GOM-001",  "Gomme Blanche",  "Effaçage", 0.20, 0.70,  60, 10),
            Produit("PAP-A4",   "Rame Papier A4", "Papier",   2.50, 5.00, 300, 50),
            Produit("PAP-A3",   "Rame Papier A3", "Papier",   4.00, 8.00,   6, 10),
            Produit("CIS-001",  "Ciseaux 17cm",   "Coupe",    1.50, 4.00,  25,  5),
            Produit("REG-001",  "Règle 30cm",     "Mesure",   0.80, 2.00,  40, 10),
            Produit("CAR-001",  "Carnet A5",      "Papier",   1.20, 3.50,   3, 10),
        ]
        for p in demo:
            self._produits[p.ref] = p
        self.sauvegarder()
```

### Étape 3 — Interface graphique (30 min)

L'UI utilise **customtkinter** (version enrichie de Tkinter) et **Treeview** (ttk) pour le tableau.

```python
# ui/app.py  (squelette)
import customtkinter as ctk
from services.stock_service import StockService
from ui.frames.stock_frame  import StockFrame

class AlQalamApp(ctk.CTk):
    def __init__(self):
        super().__init__()
        self.title("Al Qalam — Gestion de Stock")
        self.geometry("1100x680")
        self.stock = StockService()
        self._construire_interface()
        self.protocol("WM_DELETE_WINDOW", self._quitter)

    def _construire_interface(self):
        # En-tête bleu
        entete = ctk.CTkFrame(self, fg_color="#1F4E79", height=55, corner_radius=0)
        entete.pack(fill="x")
        ctk.CTkLabel(entete, text="📚  Al Qalam — Gestion de Stock",
                     font=ctk.CTkFont(size=20, weight="bold"),
                     text_color="white").pack(side="left", padx=20, pady=10)
        # Tableau stock
        self.stock_frame = StockFrame(self, self.stock, callbacks={...})
        self.stock_frame.pack(fill="both", expand=True)
```

```python
# ui/frames/stock_frame.py  (Treeview)
from tkinter import ttk

COLONNES = ("ref", "nom", "categorie", "prix_vente", "qte", "seuil", "statut")

class StockFrame(ctk.CTkFrame):
    def __init__(self, parent, stock_service, callbacks):
        super().__init__(parent)
        self._stock     = stock_service
        self._callbacks = callbacks
        self._construire_tableau()
        self.rafraichir()

    def _construire_tableau(self):
        self._arbre = ttk.Treeview(self, columns=COLONNES, show="headings")
        for col in COLONNES:
            self._arbre.heading(col, text=col.capitalize())
        self._arbre.pack(fill="both", expand=True)

    def rafraichir(self):
        # Vider et repeupler
        for item in self._arbre.get_children():
            self._arbre.delete(item)
        for p in self._stock.lister_tous():
            statut = "⚠️ Alerte" if p.est_en_alerte() else "✅ OK"
            self._arbre.insert("", "end", iid=p.ref,
                values=(p.ref, p.nom, p.categorie,
                        f"{p.prix_vente:.2f} TND",
                        p.qte, p.seuil_min, statut))
```
