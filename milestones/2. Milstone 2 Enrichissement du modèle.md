## Concepts mis en œuvre

| Méthode magique | Opérateur / usage | Bénéfice |
|-----------------|-------------------|----------|
| `__eq__(self, other)` | `produit_a == produit_b` | Comparaison par référence |
| `__lt__(self, other)` | `produit_a < produit_b` | Tri avec `sorted()` |
| `__hash__(self)` | `set(produits)`, dict keys | Stockage dans ensemble/dict |
| `__str__(self)` | `print(produit)` | Affichage lisible |
| `__repr__(self)` | En débogueur | Représentation technique |
| `__len__(self)` | `len(stock)` | Nombre de produits |
| `__contains__(self, ref)` | `"CRAY-001" in stock` | Test d'appartenance |
| `__iter__(self)` | `for p in stock:` | Itération sur le stock |
| `__enter__` / `__exit__` | `with FichierStock(...):` | Gestionnaire de contexte |

### Étape 1 — Méthodes magiques sur Produit (15 min)

```python
# models/produit.py  — ajouts V2

class Produit:
    # ... (tout le code V1 conservé) ...

    # ── Comparaison ───────────────────────────────────────────────────────
    def __eq__(self, other) -> bool:
        """Deux produits sont égaux si leur référence est identique."""
        if not isinstance(other, Produit):
            return NotImplemented
        return self.ref == other.ref

    def __lt__(self, other) -> bool:
        """Tri naturel : par référence alphabétique."""
        if not isinstance(other, Produit):
            return NotImplemented
        return self.ref < other.ref

    def __hash__(self) -> int:
        """Permet d'utiliser Produit dans un set ou comme clé de dict."""
        return hash(self.ref)

    # ── Représentation ────────────────────────────────────────────────────
    def __str__(self) -> str:
        statut = "⚠️ ALERTE" if self.est_en_alerte() else "✅ OK"
        return f"[{self.ref}] {self.nom} — Qté: {self.qte} / Seuil: {self.seuil_min} {statut}"

    def __repr__(self) -> str:
        return (f"Produit(ref={self.ref!r}, nom={self.nom!r}, "
                f"qte={self.qte}, seuil_min={self.seuil_min})")
```

**Pourquoi définir `__hash__` quand on définit `__eq__` ?**

Python exige que si deux objets sont égaux (`a == b`), ils aient le même hash (`hash(a) == hash(b)`). Si vous définissez `__eq__` sans `__hash__`, Python rend l'objet non-hashable (ne peut plus être mis dans un `set` ou utilisé comme clé de `dict`).

```python
# Démonstration
p1 = Produit("CRAY-001", "Crayon HB", "Écriture", 0.15, 0.50, 150, 20)
p2 = Produit("CRAY-001", "Crayon HB", "Écriture", 0.15, 0.50, 80, 20)  # qte différente

print(p1 == p2)         # True  → même référence
print(sorted([p2, p1])) # triés par ref
refs_uniques = {p1, p2} # set possible grâce à __hash__
print(len(refs_uniques)) # 1 — ils sont "égaux" pour le set
```

### Étape 2 — Gestionnaire de contexte FichierStock (15 min)

Un **gestionnaire de contexte** (`with`) garantit que des ressources sont toujours libérées, même en cas d'exception.

```python
# services/stock_service.py

class FichierStock:
    """
    Gestionnaire de contexte pour l'écriture sécurisée du JSON.
    Remplace le write_text() direct de V1 pour garantir la fermeture du fichier.
    """

    def __init__(self, chemin):
        self.chemin   = chemin
        self._fichier = None

    def __enter__(self):
        """Ouvre le fichier et le retourne pour utilisation dans le bloc with."""
        self._fichier = open(self.chemin, "w", encoding="utf-8")
        return self._fichier       # ← valeur de 'as f' dans 'with ... as f:'

    def __exit__(self, type_erreur, valeur, traceback):
        """
        Appelé TOUJOURS à la sortie du bloc with, qu'il y ait une erreur ou non.
        type_erreur, valeur, traceback : informations sur l'exception (None si pas d'erreur)
        Retourner False = ne pas étouffer l'exception.
        """
        if self._fichier:
            self._fichier.close()
        return False   # propager l'exception si elle existe

# Utilisation dans sauvegarder()
def sauvegarder(self) -> None:
    DATA_DIR.mkdir(parents=True, exist_ok=True)
    data = [p.to_dict() for p in self._produits.values()]
    with FichierStock(self._chemin) as f:     # __enter__ ouvre, __exit__ ferme
        f.write(json.dumps(data, indent=2, ensure_ascii=False))
```

**Comparaison V1 vs V2 :**

```python
# V1 : risque de fichier non fermé si erreur
self._chemin.write_text(json.dumps(data), encoding="utf-8")

# V2 : fermeture garantie même si json.dumps() lève une exception
with FichierStock(self._chemin) as f:
    f.write(json.dumps(data, indent=2, ensure_ascii=False))
```

### Étape 3 — Méthodes magiques sur StockService (10 min)

```python
# services/stock_service.py  — ajouts V2

class StockService:
    # ... (code V1 conservé) ...

    def __len__(self) -> int:
        """len(stock) → nombre de produits en catalogue."""
        return len(self._produits)

    def __contains__(self, ref: str) -> bool:
        """"CRAY-001" in stock → True si le produit existe."""
        return ref in self._produits

    def __iter__(self):
        """for produit in stock: ... → itère sur tous les produits."""
        return iter(self._produits.values())

    def __str__(self) -> str:
        """print(stock) affiche un résumé propre."""
        alertes = len(self.produits_en_alerte())
        return f"Al Qalam Stock · {len(self)} produits · {alertes} alerte(s)"
```

```python
# Démonstration des méthodes magiques
stock = StockService()

print(len(stock))              # 10
print("CRAY-001" in stock)     # True
print("XXXX-999" in stock)     # False

for produit in stock:          # __iter__
    if produit.est_en_alerte():
        print(produit)         # __str__ de Produit

# Tri avec __lt__ de Produit
produits_tries = sorted(stock)  # stock est itérable → sorted() fonctionne
```

### Étape 4 — Tri par colonne dans le tableau (15 min)

Cliquer sur l'en-tête d'une colonne trie le tableau dans l'ordre croissant / décroissant.

```python
# ui/frames/stock_frame.py

class StockFrame(ctk.CTkFrame):
    def __init__(self, parent, stock, callbacks):
        super().__init__(parent)
        self._tri_colonne  = None
        self._tri_inverse  = False
        self._construire_tableau()

    def _construire_tableau(self):
        self._arbre = ttk.Treeview(self, columns=COLONNES, show="headings")
        for col in COLONNES:
            # Chaque en-tête déclenche le tri sur sa colonne
            self._arbre.heading(col, text=col.capitalize(),
                                command=lambda c=col: self._trier_par(c))

    def _trier_par(self, colonne: str) -> None:
        """Trie le tableau par la colonne cliquée, inverse à chaque clic."""
        if colonne == self._tri_colonne:
            self._tri_inverse = not self._tri_inverse
        else:
            self._tri_colonne = colonne
            self._tri_inverse = False
        self.rafraichir()   # reconstruit le tableau trié
```

### Étape 5 — Fiche détail et modification (15 min)

```python
# ui/frames/dialogs.py

class DialogueFicheDetail(ctk.CTkToplevel):
    """Fenêtre modale en lecture seule — double-clic sur un produit."""

    def __init__(self, parent, produit):
        super().__init__(parent)
        self.title(f"Fiche — {produit.nom}")
        self.resizable(False, False)
        self._afficher(produit)

    def _afficher(self, p):
        champs = [
            ("Référence",   p.ref),
            ("Nom",         p.nom),
            ("Catégorie",   p.categorie),
            ("Prix achat",  f"{p.prix_achat:.3f} TND"),
            ("Prix vente",  f"{p.prix_vente:.3f} TND"),
            ("Marge",       f"{p.marge_unitaire():.3f} TND"),
            ("Quantité",    str(p.qte)),
            ("Seuil min",   str(p.seuil_min)),
            ("Valeur stock",f"{p.valeur_stock():.2f} TND"),
            ("Statut",      "⚠️ ALERTE" if p.est_en_alerte() else "✅ OK"),
        ]
        for label, valeur in champs:
            ligne = ctk.CTkFrame(self, fg_color="transparent")
            ligne.pack(fill="x", padx=20, pady=2)
            ctk.CTkLabel(ligne, text=label + " :", width=120,
                         anchor="w", font=ctk.CTkFont(weight="bold")).pack(side="left")
            ctk.CTkLabel(ligne, text=valeur, anchor="w").pack(side="left")
```
