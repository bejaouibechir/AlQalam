## Etape 1 -- Decorateur simple : @chronometre

```python
import functools, time

def chronometre(func):
    @functools.wraps(func)   # TOUJOURS : preserve __name__, __doc__, __qualname__
    def wrapper(*args, **kwargs):
        debut    = time.perf_counter()
        resultat = func(*args, **kwargs)
        print(f"[chrono] {func.__qualname__} -> {(time.perf_counter()-debut)*1000:.2f} ms")
        return resultat
    return wrapper
```

Pourquoi @functools.wraps est obligatoire :

```python
# Sans @wraps
@chronometre
def sauvegarder(self): "Sauvegarde le stock."

print(sauvegarder.__name__)  # "wrapper"        INCORRECT
print(sauvegarder.__doc__)   # None             INCORRECT

# Avec @functools.wraps(func)
print(sauvegarder.__name__)  # "sauvegarder"    OK
print(sauvegarder.__doc__)   # "Sauvegarde..."  OK
```

## Etape 2 -- Decorateur avec parametre : @journaliser("entree")

```python
@journaliser("entree")
def entree_stock(self, ref, qte, note=""):
    ...

# Equivalent exact :
# entree_stock = journaliser("entree")(entree_stock)
```

```python
def journaliser(operation: str):        # NIVEAU 1 -- fabrique, recoit "entree"

    def decorateur(func):               # NIVEAU 2 -- recoit la fonction decoree

        @functools.wraps(func)
        def wrapper(self, *args, **kwargs):  # NIVEAU 3 -- execute a chaque appel
            debut = datetime.now()
            try:
                resultat = func(self, *args, **kwargs)
                return resultat
            except Exception:
                raise
            finally:
                # finally : s'execute TOUJOURS, meme si exception levee
                duree_ms = int((datetime.now() - debut).total_seconds() * 1000)
                if hasattr(self, "_journal"):
                    self._journal.enregistrer(operation=operation, ...)

        return wrapper

    return decorateur   # la fabrique retourne le vrai decorateur
```

## Etape 3 -- Double decoration et ordre d'execution

```python
@journaliser("entree")       # decorateur EXTERNE (enveloppe externe)
@valider_qte(min_val=1)      # decorateur INTERNE (enveloppe interne)
def entree_stock(self, ref, qte, note=""):
    ...
```

Ordre a l'appel avec qte=50 valide :
  1. wrapper_journaliser demarre
  2.   wrapper_valider_qte valide qte=50
  3.     entree_stock s'execute
  2.   wrapper_valider_qte retourne
  1. wrapper_journaliser enregistre SUCCES

Ordre a l'appel avec qte=-5 invalide :
  1. wrapper_journaliser demarre
  2.   wrapper_valider_qte leve ValueError
  1. wrapper_journaliser finally -> enregistre ECHEC
  -> ValueError propagee vers l'UI

Le journal capture toutes les tentatives, valides ou non.

## Etape 4 -- Protocole descripteur sur Produit

```python
class Positif:
    def __set_name__(self, owner, name):
        # Appele une fois a la construction de la classe
        self._nom_prive = f"_{name}"   # "prix_achat" -> "_prix_achat"

    def __get__(self, obj, objtype=None):
        if obj is None: return self   # acces via Produit.prix_achat
        return getattr(obj, self._nom_prive, 0)

    def __set__(self, obj, valeur):
        # Appele a CHAQUE ecriture : produit.prix_achat = 1.5
        valeur = float(valeur)
        if valeur < 0:
            raise ValueError(f"Valeur negative interdite : {valeur}")
        setattr(obj, self._nom_prive, valeur)


class Produit:
    # Descripteurs au niveau de la CLASSE -- interceptent toute affectation
    prix_achat = Positif()
    prix_vente = Positif()
    qte        = PositifEntier()
    seuil_min  = PositifEntier()
    ref        = NonVide()
    nom        = NonVide()
    categorie  = NonVide()
```

Avantage vs @property : un descripteur est reutilisable sur N attributs de N classes.
Avec @property il faut copier le code pour chaque attribut.

## Etape 5 -- @dataclass pour EntreeJournal

```python
from dataclasses import dataclass, field
from datetime import datetime

@dataclass
class EntreeJournal:
    # @dataclass genere __init__, __repr__, __eq__ automatiquement
    operation : str
    ref       : str
    succes    : bool
    erreur    : str    = ""
    duree_ms  : int    = 0
    # field(default_factory=...) : valeur calculee a chaque instanciation
    timestamp : datetime = field(default_factory=datetime.now)

e = EntreeJournal("entree", "CRAY-001", True, duree_ms=3)
print(e)
# EntreeJournal(operation='entree', ref='CRAY-001', succes=True, duree_ms=3, ...)
```
