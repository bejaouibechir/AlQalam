## 🔧 Partie 1 — Le module `validators/`

### `regex_validators.py` — patterns compilés

```python
import re

REGEX = {
    # Référence produit : 2-6 MAJ + tiret + 1-5 alphanum MAJ
    # CRAY-001  PAP-A4  STYL-002  REG-001
    "ref": re.compile(r"^[A-Z]{2,6}-[A-Z0-9]{1,5}$"),

    # Nom : 2 à 60 caractères
    "nom": re.compile(r"^.{2,60}$"),

    # Prix TND : entier ou décimal (virgule ou point, 1-3 décimales)
    "prix": re.compile(r"^\d+([.,]\d{1,3})?$"),

    # Quantité / seuil : entier ≥ 0
    "qte": re.compile(r"^\d+$"),

    # Note : optionnelle, max 200 caractères
    "note": re.compile(r"^.{0,200}$", re.DOTALL),
}
```

**Pourquoi `re.compile()` ?**

```python
# Sans compilation — Python reparse le pattern à CHAQUE appel
# Problème : <KeyRelease> est déclenché à chaque touche → appels très fréquents

for frappe in frappes_utilisateur:
    result = re.match(r"^[A-Z]{2,6}-[A-Z0-9]{1,5}$", frappe)  # reparse à chaque fois ❌

# Avec compilation — pattern analysé UNE SEULE FOIS
REF_RE = re.compile(r"^[A-Z]{2,6}-[A-Z0-9]{1,5}$")

for frappe in frappes_utilisateur:
    result = REF_RE.match(frappe)   # réutilise la même regex compilée ✅
```

### Fonctions de validation

Chaque fonction retourne un tuple `(valide: bool, message: str)` :

```python
def valider_ref(texte: str) -> tuple[bool, str]:
    t = texte.strip().upper()
    if not t:
        return False, "La référence est obligatoire."
    if REGEX["ref"].match(t):
        return True, "✓ Référence valide"
    return False, "Format : 2-6 lettres MAJ + tiret + 1-5 alphanum (ex: CRAY-001)"

def valider_prix(texte: str) -> tuple[bool, str]:
    t = texte.strip()
    if not t:
        return False, "Le prix est obligatoire."
    if REGEX["prix"].match(t):
        return True, "✓ Prix valide"
    return False, "Format attendu : nombre entier ou décimal (ex: 1.500 ou 12,90)"
```

**Tests de la regex référence :**

```python
from validators.regex_validators import valider_ref

print(valider_ref("CRAY-001"))   # (True,  "✓ Référence valide")
print(valider_ref("PAP-A4"))     # (True,  "✓ Référence valide")
print(valider_ref("cray-001"))   # (False, "Format : ...")  ← minuscules rejetées
print(valider_ref("REF-123456")) # (False, "Format : ...")  ← 6 chiffres > max 5
print(valider_ref(""))           # (False, "La référence est obligatoire.")
```

## 🔧 Partie 3 — `LogParser` : analyse du journal par regex

### Flux de données

```
JournalService.get_entrees()
        │
        ▼
LogParser.formater_journal()     → texte multiligne structuré
        │
        ▼
LogParser.analyser()             → StatutLog (stats extraites par regex)
        │
        ├── re.findall(r"✅", texte)         → nb_succes
        ├── re.findall(r"❌", texte)         → nb_erreurs
        ├── re.finditer(pattern_op, texte)   → par_operation (Counter)
        └── REGEX["ref_dans_texte"].findall  → refs_actives (top 5)
```

### Format du log généré

```
[2025-03-09 10:42:01] [entree   ] ✅ CRAY-001   | Entrée stock    | +50 unités
[2025-03-09 10:43:15] [sortie   ] ✅ STYL-001   | Sortie stock    | -5 unités
[2025-03-09 10:44:00] [ajout    ] ✅ NEW-001    | Nouveau produit | ref=NEW-001
[2025-03-09 10:45:00] [sortie   ] ❌ REG-001    | Sortie stock    | stock insuffisant
```

### Méthode `analyser()` — regex en action

```python
def analyser(self) -> StatutLog:
    texte = self.formater_journal()

    # re.findall() — trouve TOUTES les occurrences d'un pattern
    nb_succes  = len(re.findall(r"✅", texte))
    nb_erreurs = len(re.findall(r"❌", texte))

    # re.findall() + re.MULTILINE — lignes contenant ❌
    # ^ et $ ancrent sur chaque LIGNE grâce à re.MULTILINE
    lignes_erreur = re.findall(r"^.*❌.*$", texte, re.MULTILINE | re.UNICODE)

    # re.finditer() avec groupe nommé (?P<op>...) — extraire les types d'opération
    pattern_op = re.compile(r"\[(?P<op>[a-z]+)\s*\]", re.IGNORECASE)
    operations = [m.group("op") for m in pattern_op.finditer(texte)]
    par_operation = Counter(operations)  # {"entree": 5, "sortie": 3, ...}

    # Pattern pré-compilé — REGEX["ref_dans_texte"] = r"\b[A-Z]{2,6}-[A-Z0-9]{1,5}\b"
    toutes_refs   = REGEX["ref_dans_texte"].findall(texte)
    refs_actives  = [ref for ref, _ in Counter(toutes_refs).most_common(5)]

    return StatutLog(
        nb_total=nb_succes + nb_erreurs,
        nb_erreurs=nb_erreurs,
        nb_succes=nb_succes,
        par_operation=par_operation,
        refs_actives=refs_actives,
        lignes_erreur=lignes_erreur,
        texte_brut=texte,
    )
```

### Recherche interactive par regex

```python
def rechercher(self, motif: str, ignorer_casse: bool = True) -> list[str]:
    """
    Filtre les lignes du journal par regex.
    re.MULTILINE : ^ et $ ancrent sur chaque ligne du texte.
    """
    texte = self.formater_journal()
    flags = re.MULTILINE | re.UNICODE
    if ignorer_casse:
        flags |= re.IGNORECASE

    try:
        # Construit un pattern qui sélectionne les lignes contenant le motif
        pattern = re.compile(rf"^.*{motif}.*$", flags)
        return pattern.findall(texte)
    except re.error:
        # Motif invalide (ex: "*abc") → affiche tout sans planter
        return texte.splitlines()
```

**Exemples de recherches :**

```python
parser.rechercher("CRAY")          # lignes contenant "CRAY" (insensible casse)
parser.rechercher("CRAY|STYL")     # lignes contenant "CRAY" OU "STYL" — alternance regex
parser.rechercher("entree")        # lignes de type entrée
parser.rechercher(r"[-]\d+")       # lignes avec une quantité négative (sorties)
parser.rechercher("❌", ignorer_casse=False)  # uniquement les lignes en erreur
```

def _recherche_live(self, _event=None):
    motif = self.e_recherche.get().strip()
    ignorer_casse = bool(self.chk_casse.get())

    # Vérifier la validité syntaxique de la regex AVANT de l'utiliser
    try:
        re.compile(motif)
        self.lbl_regex_statut.configure(text="✓ regex valide", text_color="#27AE60")
    except re.error as exc:
        self.lbl_regex_statut.configure(
            text=f"⚠ regex invalide : {exc}", text_color="#E74C3C",
        )
        return   # ← ne pas appliquer une regex cassée

    lignes = self._parser.rechercher(motif, ignorer_casse=ignorer_casse)
    self._afficher_lignes(lignes)

## 💡 Points clés à retenir

> **`re.compile()`** : toujours compiler les patterns réutilisés — essentiel pour les callbacks `<KeyRelease>` qui s'exécutent à chaque frappe.

> **`re.MULTILINE`** : sans ce flag, `^` et `$` ancrent sur toute la chaîne — avec ce flag, ils ancrent sur chaque ligne. Indispensable pour filtrer des logs multiligne.

> **`re.findall()` vs `re.finditer()`** : `findall` retourne une liste de chaînes (pratique pour compter), `finditer` retourne des Match objects (indispensable pour accéder aux groupes nommés `(?P<nom>...)`).

> **`re.error`** : toujours capturer l'exception quand la regex vient de l'utilisateur — une regex malformée planterait sinon l'application.

> **Pattern `rf"^.*{motif}.*$"`** : f-string + raw string combinés — insère le motif utilisateur dans la regex dynamiquement.
